# 0206.反转链表

一、迭代法

需要特判数组为空，1个元素，2个元素的情况

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

//  
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(!head) return head;

        ListNode *p = head;
        ListNode *q = p->next;
        if(!q) return head;
        head->next = nullptr;
        ListNode*t = q->next;

        if(!t) {
            q->next = p;
            p->next = nullptr;
            return q;
        }

        while(t->next){
            q->next = p;
            p = q;
            q = t;
            t = t->next;
        }
        t->next = q;
        q->next = p;
        return t;
    }
};
```

二、递归

`reverseList(head)`函数功能，反转以head为头的链表，并且返回新链表的头节点。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

//  需要特判数组为空，1个元素，2个元素的情况
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(!head || !head->next) return head;
        ListNode * last = reverseList(head->next);

        head->next->next = head;
        head->next = nullptr;
        return last;
    }
};
```
